class APIConfigError(Exception):
    def __init__(
        self,
        message: str,
        service: str | None = None,
        env_var: str | None = None,
    ) -> None:
        self.message = message
        self.service = service
        self.env_var = env_var

        details = [message]
        if service:
            details.append(f"service={service}")
        if env_var:
            details.append(f"env_var={env_var}")

        super().__init__(" | ".join(details))

    def __str__(self) -> str:
        return self.args[0]


class APIRequestError(Exception):
    def __init__(
        self,
        message: str,
        service: str | None = None,
        url: str | None = None,
        status_code: int | None = None,
        response_text: str | None = None,
    ) -> None:
        self.message = message
        self.service = service
        self.url = url
        self.status_code = status_code
        self.response_text = response_text

        details = [message]
        if service:
            details.append(f"service={service}")
        if url:
            details.append(f"url={url}")
        if status_code is not None:
            details.append(f"status_code={status_code}")
        if response_text:
            details.append(f"response={response_text[:300]}")

        super().__init__(" | ".join(details))

    def __str__(self) -> str:
        return self.args[0]


class DefiLlamaAPI:
    def __init__(self) -> None:
        self.base_url = os.getenv(
            "DEFILLAMA_BASE_URL", "https://pro-api.llama.fi").rstrip("/")
        self.api_key = os.getenv("DEFILLAMA_API_KEY", "").strip()
        self.timeout = int(os.getenv("REQUEST_TIMEOUT", "20"))
        self.user_agent = os.getenv("HTTP_USER_AGENT", "CriptoBot/1.0")

        self.session = requests.Session()
        self.session.headers.update({
            "Accept": "application/json",
            "User-Agent": self.user_agent,
        })

    def _build_url(self, endpoint: str, pro: bool = False) -> str:
        endpoint = endpoint.lstrip("/")
        if pro:
            if not self.api_key:
                raise APIConfigError(
                    message="DEFILLAMA_API_KEY não configurada para endpoint Pro",
                    service="defillama",
                    env_var="DEFILLAMA_API_KEY",
                )
            return f"{self.base_url}/{self.api_key}/{endpoint}"
        return f"{self.base_url}/{endpoint}"

    def _request(
        self,
        method: str,
        endpoint: str,
        *,
        pro: bool = False,
        params: Optional[Dict[str, Any]] = None,
        json_body: Optional[Dict[str, Any]] = None,
    ) -> Dict[str, Any] | list | float | int:
        url = self._build_url(endpoint, pro=pro)
        try:
            response = self.session.request(
                method=method,
                url=url,
                params=params,
                json=json_body,
                timeout=self.timeout,
            )
            response.raise_for_status()

            content_type = response.headers.get("Content-Type", "")
            if "application/json" in content_type:
                return response.json()

            text = response.text.strip()
            try:
                return response.json()
            except Exception:
                try:
                    return float(text)
                except ValueError:
                    return text

        except requests.HTTPError as e:
            raise APIRequestError(
                message="Erro HTTP na API DefiLlama",
                service="defillama",
                url=url,
                status_code=e.response.status_code if e.response else None,
                response_text=e.response.text if e.response else None,
            ) from e
        except requests.RequestException as e:
            raise APIRequestError(
                message="Erro de requisição na API DefiLlama",
                service="defillama",
                url=url,
            ) from e
